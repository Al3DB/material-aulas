def setup():
    size(600, 600)   # área de desenho
    background(128)  # cor de fundo
    
    fill(255, 255, 0) # preenchimento atinge o texto
    textSize(60)  # tamanho em pontos
    
    margem = 100  # x inicial para desenhar afastado da borda
    pos_y = 100
    
    frutas = ["uva", "jaca", "banana", "kiwi", "pequi"]
    
    for f in frutas:
        text(f, margem, pos_y)  # testo, x, y
        pos_y = pos_y + 60   #  aumente o valor de pos_y em 20
    margem = 100  # x inicial para desenhar afastado da borda
    pos_y = 100
    

    for i in range(3, 9, 2):
        text(i, margem, pos_y)  # testo, x, y
        pos_y = pos_y + 40   #  aumente o valor de pos_y em 20
